Quel est le type de référence de ILevel ?


public class World1 implements ILevel
{
    private String n;

    public World1(String name)
    {
        this.n=name;
    }

    @Override
    public void setMapLevel(Server serv)
    {
        Grassland grassLevel = new Grassland();
        grassLevel.piece1(0, 80, Pivot.NONE);
        grassLevel.piece2(80, 160, Pivot.LEFT90);
        grassLevel.piece3(160, 240, Pivot.RIGHT90);
        serv.setWorld(grassLevel);
    }
}
